**Step A — Create `src/components/explore/format-content-wrapper.tsx` (Client Component)**

This component receives ALL compiled format components from the server, manages `activeTab` state, and wires `onFormatChange` to the state setter so tab clicks actually switch content.

```typescript
'use client'

import { useState } from 'react'
import type { ComponentType } from 'react'
import { FormatTabBar } from '@/components/explore/format-tab-bar'
import { CodeBlock } from '@/components/explore/code-block'
import { CopyButton } from '@/components/explore/copy-button'

type Format = 'framer' | 'webflow' | 'html' | 'jsx' | 'tsx'

interface FormatContentWrapperProps {
  formats: Format[]
  compiledFormats: Record<Format, { Component: ComponentType; rawSource: string }>
}

export function FormatContentWrapper({ formats, compiledFormats }: FormatContentWrapperProps) {
  const [activeFormat, setActiveFormat] = useState<Format>(formats[0])

  const active = compiledFormats[activeFormat]

  return (
    <div className="flex flex-col gap-4">
      <FormatTabBar
        formats={formats}
        activeFormat={activeFormat}
        onFormatChange={(fmt) => setActiveFormat(fmt as Format)}
        tabPanelId="code-panel"
      />

      <div
        id="code-panel"
        role="tabpanel"
        aria-label={`Kode ${activeFormat.toUpperCase()}`}
        className="relative"
      >
        {active ? (
          <>
            <div className="absolute top-3 right-3 z-10">
              <CopyButton text={active.rawSource} />
            </div>
            <CodeBlock Content={active.Component} />
          </>
        ) : (
          <div className="rounded-xl border border-neutral-200 bg-neutral-50 p-6 text-sm text-neutral-500">
            Format file not found.
          </div>
        )}
      </div>
    </div>
  )
}
```

**MDX format rendering strategy (RESEARCH.md Pattern 5 — extended to ALL formats):**
Use `Promise.all` over `resource.available_formats` to compile every format's MDX file at build/render time (inside `generateStaticParams`, the page is statically generated). Pass the resulting `Record<Format, { Component, rawSource }>` to `<FormatContentWrapper>` which manages `activeTab` state client-side. Tab clicks call `setActiveFormat` — content actually switches. This fully satisfies D-08.

If `resource.demo_url` is non-null, render a button-styled anchor `href={resource.demo_url}` with `target="_blank" rel="noopener noreferrer"` labeled "Lihat Demo". import * as runtime from 'react/jsx-runtime'
import rehypePrettyCode from 'rehype-pretty-code'
import fs from 'fs'
import path from 'path'
import type { ComponentType } from 'react'
import { getAllResources, getResourceBySlug } from '@/lib/content'

import { Badge } from '@/components/ui/badge'

type Format = 'framer' | 'webflow' | 'html' | 'jsx' | 'tsx'

// generateStaticParams runs at build time — pre-renders all resource slugs
export async function generateStaticParams() {
  const resources = await getAllResources()
  return resources.map((r) => ({ slug: r.slug }))
}

export async function generateMetadata({
  params,
}: {
  params: Promise<{ slug: string }>
}): Promise<Metadata> {
  const { slug } = await params
  const resource = await getResourceBySlug(slug)
  if (!resource) return { title: 'Komponen tidak ditemukan — lailit.supply' }
  return {
    title: `${resource.title} — lailit.supply`,
    description: `Komponen ${resource.title} untuk ${resource.available_formats.join(', ')}.`,
  }
}

// Compile a single format MDX file using Pattern 5 (programmatic compile + run).
// Returns null if the file does not exist.
async function compileFormat(
  slug: string,
  format: Format
): Promise<{ Component: ComponentType; rawSource: string } | null> {
  const contentRoot = path.join(process.cwd(), 'content/resources')
  const formatFilePath = path.join(contentRoot, slug, `${format}.mdx`)
  if (!fs.existsSync(formatFilePath)) return null

  const source = fs.readFileSync(formatFilePath, 'utf-8')
  const compiled = await compile(source, {
    outputFormat: 'function-body',
    rehypePlugins: [[rehypePrettyCode, { theme: 'github-dark-dimmed' }]],
  })
  const mod = await run(compiled, { ...runtime, baseUrl: import.meta.url })
  return { Component: mod.default as ComponentType, rawSource: source }
}

export default async function ResourcePage({
  params,
}: {
  params: Promise<{ slug: string }>  // MUST be Promise type — Next.js 16 breaking change
}) {
  const { slug } = await params  // MUST await — sync access throws TypeError

  const resource = await getResourceBySlug(slug)
  if (!resource) notFound()

  // D-09: Premium resources — return paywall stub, no server-side content rendering
  if (resource.is_premium) {
    return (
      <div className="px-4 sm:px-8 lg:px-12 py-16 lg:py-24">
        <div className="max-w-[1200px] mx-auto">
          <Link
            href="/explore"
            className="inline-flex items-center text-sm text-neutral-500 hover:text-neutral-950 transition-colors duration-150 mb-8 focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-neutral-950"
          >
            ← Kembali ke Jelajahi
          </Link>
          <div className="flex flex-col lg:flex-row gap-12">
            <div className="lg:w-[60%]">
              <ThumbnailPlaceholder category={resource.category} title={resource.title} />
              <h1 className="mt-6 text-3xl font-semibold leading-[1.2] tracking-[-0.02em] text-neutral-950">
                {resource.title}
              </h1>
              <div className="mt-3 flex gap-2 items-center">
                <Badge variant="outline" className="text-xs">Premium</Badge>
                <span className="text-sm text-neutral-500">{resource.category}</span>
              </div>
            </div>
            <div className="lg:w-[40%]">
              <PaywallStub />
            </div>
          </div>
        </div>
      </div>
    )
  }

  // Free resource: compile ALL available formats at build time (Pattern 5 — extended).
  // Promise.all over available_formats produces a Record<Format, { Component, rawSource }>.
  // FormatContentWrapper (Client Component) manages activeTab state and switches content.
  const compiledEntries = await Promise.all(
    resource.available_formats.map(async (fmt) => {
      const result = await compileFormat(slug, fmt as Format)
      return [fmt, result] as [Format, { Component: ComponentType; rawSource: string } | null]
    })
  )

  const compiledFormats = Object.fromEntries(
    compiledEntries.filter((e): e is [Format, { Component: ComponentType; rawSource: string }] => e[1] !== null)
  ) as Record<Format, { Component: ComponentType; rawSource: string }>

    animation: 'Animation',
    'ui-components': 'UI Components',
    layout: 'Layout',
    interactions: 'Interactions',
  }

  return (
    <div className="px-4 sm:px-8 lg:px-12 py-16 lg:py-24">
      <div className="max-w-[1200px] mx-auto">
        {/* Back link */}
        <Link
          href="/explore"
          className="inline-flex items-center text-sm text-neutral-500 hover:text-neutral-950 transition-colors duration-150 mb-8 focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-neutral-950"
        >
          ← Kembali ke Jelajahi
        </Link>

        <div className="flex flex-col lg:flex-row gap-12">
          {/* Left column — media + meta */}
          <div className="lg:w-[60%] flex flex-col gap-6">
            <VideoPlaceholder title={resource.title} />
            <div>
              <h1 className="text-3xl font-semibold leading-[1.2] tracking-[-0.02em] text-neutral-950">
                {resource.title}
              </h1>
              <div className="mt-3 flex gap-2 items-center flex-wrap">
                <Badge variant="default" className="text-xs bg-neutral-950 text-white">
                  Gratis
                </Badge>
                <span className="text-sm text-neutral-500">
                  {categoryLabels[resource.category] ?? resource.category}
                </span>
              </div>
              {/* demo_url: render "Lihat Demo" link only when non-null */}
              {resource.demo_url && (
                <a
                  href={resource.demo_url}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="mt-4 inline-flex items-center gap-1.5 rounded-lg border border-neutral-200 bg-white px-4 py-2 text-sm font-medium text-neutral-700 hover:bg-neutral-50 hover:text-neutral-950 transition-colors duration-150 focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-neutral-950"
