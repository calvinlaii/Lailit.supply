```typescript
export const getMembership = cache(async () => {
  const supabase = await createClient()
  const {
    data: { user },
  } = await supabase.auth.getUser()
  if (!user) return null

  const { data, error } = await supabase
    .from('users')
    .select(
      'membership_tier, membership_status, membership_expires_at, lifetime_purchased, mayar_member_id'
    )
    .eq('id', user.id)
    .single()

  if (error) return null
  return data as {
    membership_tier: 'monthly' | 'lifetime' | null
    membership_status: 'active' | 'canceled' | 'expired' | null
    membership_expires_at: string | null
    lifetime_purchased: boolean
    mayar_member_id: string | null
  }
})
```

**Part A — Create `src/components/account/account-membership-card.tsx`:**

```typescript
import { Badge } from '@/components/ui/badge'
import { Button } from '@/components/ui/button'
import {
  Card,
  CardContent,
  CardFooter,
  CardHeader,
} from '@/components/ui/card'

type MembershipData = {
  membership_tier: 'monthly' | 'lifetime' | null
  membership_status: 'active' | 'canceled' | 'expired' | null
  membership_expires_at: string | null
  lifetime_purchased: boolean
  mayar_member_id: string | null
} | null

interface AccountMembershipCardProps {
  membership: MembershipData
  customerPortalUrl: string | undefined
}

// Format a timestamptz string as Bahasa Indonesia date
function formatDate(isoString: string): string {
  return new Intl.DateTimeFormat('id-ID', {
    day: 'numeric',
    month: 'long',
    year: 'numeric',
  }).format(new Date(isoString))
}

export function AccountMembershipCard({
  membership,
  customerPortalUrl,
}: AccountMembershipCardProps) {
  const isLifetime = membership?.lifetime_purchased === true
  const tier = membership?.membership_tier
  const status = membership?.membership_status

  // Plan label
  const planLabel = isLifetime || tier === 'lifetime' ? 'Seumur Hidup' : tier === 'monthly' ? 'Bulanan' : null

  // Status badge
  type BadgeVariant = 'default' | 'secondary' | 'destructive' | 'outline'
  const statusBadge: { variant: BadgeVariant; label: string } = (() => {
    if (!status) return { variant: 'outline', label: 'Belum berlangganan' }
    if (status === 'active' && isLifetime) return { variant: 'default', label: 'Aktif — Seumur Hidup' }
    if (status === 'active') return { variant: 'default', label: 'Aktif' }
    if (status === 'canceled') return { variant: 'outline', label: 'Dibatalkan' }
    return { variant: 'destructive', label: 'Kedaluwarsa' }
  })()

  // Expiry display
  const expiryDisplay = (() => {
    if (isLifetime) return 'Akses Seumur Hidup'
    if (!membership?.membership_expires_at) return '—'
    return `Berlaku hingga ${formatDate(membership.membership_expires_at)}`
  })()

  // Grace period note — only for canceled users with an expiry date
  const showGraceNote =
    status === 'canceled' && membership?.membership_expires_at != null
  const graceNote = showGraceNote
    ? `Akses aktif sampai ${formatDate(membership!.membership_expires_at!)}`
    : null

  // Empty state — no membership at all
  if (!planLabel) {
    return (
      <Card className="max-w-[480px]">
        <CardHeader>
          <h2 className="text-base font-semibold leading-[1.5] text-neutral-950">
            Belum berlangganan
          </h2>
        </CardHeader>
        <CardContent>
          <p className="text-sm font-normal leading-[1.45] text-neutral-500">
            Pilih paket untuk mulai mengakses semua komponen premium.
          </p>
        </CardContent>
        <CardFooter>
          <Button asChild variant="default" className="w-full">
            <a href="/pricing">Lihat Paket</a>
          </Button>
        </CardFooter>
      </Card>
    )
  }

  return (
    <Card className="max-w-[480px]">
      <CardHeader>
        <div className="flex items-center justify-between gap-4">
          <h2 className="text-base font-semibold leading-[1.5] text-neutral-950">
            {planLabel}
          </h2>
          <Badge variant={statusBadge.variant}>{statusBadge.label}</Badge>
        </div>
      </CardHeader>
      <CardContent className="flex flex-col gap-2">
        <p className="text-sm font-normal leading-[1.45] text-neutral-500">
          {expiryDisplay}
        </p>
        {graceNote && (
          <p className="text-sm font-normal leading-[1.45] text-neutral-500">
            {graceNote}
          </p>
        )}
      </CardContent>
      <CardFooter>
        <Button
          asChild
          variant="default"
          className="w-full"
        >
          <a
            href={customerPortalUrl ?? '#'}
            target="_blank"
            rel="noopener noreferrer"
            aria-label="Kelola langganan di Mayar — tautan eksternal"
          >
            Kelola Langganan
          </a>
        </Button>
      </CardFooter>
    </Card>
  )
}
```

|-----------|----------|-----------|-------------|-----------------|
