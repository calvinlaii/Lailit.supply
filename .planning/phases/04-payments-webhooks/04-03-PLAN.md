```typescript
import type { NextRequest } from 'next/server'
import { createClient } from '@/lib/supabase/server'
import { createAdminClient } from '@/lib/supabase/admin'
import { Resend } from 'resend'
import { WelcomeEmail } from '@/emails/WelcomeEmail'

// ─── Types ────────────────────────────────────────────────────────────────────

interface MayarWebhookPayload {
  'event.received': string
  data: {
    id: string
    status: boolean
    customerName: string
    customerEmail: string
    productName: string
    productId: string
    merchantId: string
    amount: number
    createdAt: string
    updatedAt: string
  }
}

// ─── Cross-verify ─────────────────────────────────────────────────────────────

async function crossVerifyWithMayar(
  transactionId: string
): Promise<{ valid: boolean; httpStatus: number }> {
  const baseUrl = process.env.MAYAR_API_BASE_URL ?? 'https://api.mayar.id'
  let res: Response
  try {
    res = await fetch(`${baseUrl}/hl/v1/payment/${transactionId}`, {
      headers: {
        Authorization: `Bearer ${process.env.MAYAR_API_KEY}`,
      },
      cache: 'no-store',
    })
  } catch {
    // Network error (Mayar unreachable) — return 503 so Mayar retries
    return { valid: false, httpStatus: 503 }
  }

  if (res.status === 503 || res.status === 502 || res.status === 504) {
    return { valid: false, httpStatus: 503 }
  }
  if (!res.ok) {
    // 404 or other error — transaction not found (forged webhook)
    return { valid: false, httpStatus: 400 }
  }

  let json: { statusCode?: number; data?: unknown }
  try {
    json = await res.json()
  } catch {
    return { valid: false, httpStatus: 400 }
  }

  if (json.statusCode !== 200 || !json.data) {
    return { valid: false, httpStatus: 400 }
  }

  return { valid: true, httpStatus: 200 }
}

// ─── Tier detection ───────────────────────────────────────────────────────────

function detectTier(productName: string): 'monthly' | 'lifetime' {
  // productName from Mayar payload (e.g. "Bulanan", "Lifetime", "Seumur Hidup")
  // Case-insensitive match for lifetime keywords
  const lc = productName.toLowerCase()
  if (lc.includes('lifetime') || lc.includes('seumur hidup')) {
    return 'lifetime'
  }
  return 'monthly'
}

// ─── Event handlers ───────────────────────────────────────────────────────────

async function handleNewMember(body: MayarWebhookPayload): Promise<void> {
  const { customerEmail, customerName, productName, id: mayarEventId } = body.data
  const admin = createAdminClient()
  const tier = detectTier(productName)

  // Step 1: Find or create Supabase auth user (Pitfall 5 — may already exist on re-sub)
  let userId: string
  const { data: existingUserData } = await admin.auth.admin.getUserByEmail(customerEmail)

  if (existingUserData?.user) {
    userId = existingUserData.user.id
  } else {
    const { data: newUserData, error: createError } = await admin.auth.admin.createUser({
      email: customerEmail,
      email_confirm: true,
      user_metadata: { full_name: customerName },
    })
    if (createError || !newUserData?.user) {
      throw new Error(`createUser failed: ${createError?.message}`)
    }
    userId = newUserData.user.id
  }

  // Step 2: Upsert public.users membership columns
  const membershipData = {
    id: userId,
    email: customerEmail,
    membership_tier: tier,
    membership_status: 'active' as const,
    membership_expires_at: tier === 'lifetime' ? null : new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toISOString(),
    lifetime_purchased: tier === 'lifetime',
    mayar_member_id: body.data.id,
  }

  const supabase = await createClient()
  const { error: upsertError } = await supabase
    .from('users')
    .upsert(membershipData, { onConflict: 'id' })

  if (upsertError) {
    throw new Error(`users upsert failed: ${upsertError.message}`)
  }

  // Step 3: Generate magic link (lands on /dashboard)
  const { data: linkData, error: linkError } = await admin.auth.admin.generateLink({
    type: 'magiclink',
    email: customerEmail,
    options: {
      redirectTo: `${process.env.NEXT_PUBLIC_SITE_URL}/dashboard`,
    },
  })

  if (linkError || !linkData?.properties?.action_link) {
    console.error('generateLink failed:', linkError?.message)
    // Non-fatal: user account exists, email send skipped
    return
  }

  const magicLink = linkData.properties.action_link
  const planLabel = tier === 'lifetime' ? 'Seumur Hidup' : 'Bulanan'

  // Step 4: Send welcome email via Resend
  const resend = new Resend(process.env.RESEND_API_KEY)
  const { error: emailError } = await resend.emails.send({
    from: 'lailit.supply <noreply@lailit.supply>',
    to: [customerEmail],
    subject: 'Selamat datang di lailit.supply 🎉',
    react: WelcomeEmail({ name: customerName ?? '', magicLink, plan: planLabel }),
  })

  if (emailError) {
    // Log but don't throw — account creation succeeded even if email failed
    console.error('Resend email failed:', JSON.stringify(emailError))
  }
}

async function handlePaymentReceived(body: MayarWebhookPayload): Promise<void> {
  const supabase = await createClient()
  // Extend membership by 30 days from now
  const newExpiry = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toISOString()
  const { error } = await supabase
    .from('users')
    .update({ membership_expires_at: newExpiry, membership_status: 'active' })
    .eq('email', body.data.customerEmail)
  if (error) throw new Error(`payment.received update failed: ${error.message}`)
}

async function handleUnsubscribed(body: MayarWebhookPayload): Promise<void> {
  // D-07: Set status to 'canceled' — DO NOT touch membership_expires_at
  const supabase = await createClient()
  const { error } = await supabase
    .from('users')
    .update({ membership_status: 'canceled' })
    .eq('email', body.data.customerEmail)
  if (error) throw new Error(`memberUnsubscribed update failed: ${error.message}`)
}

async function handleExpired(body: MayarWebhookPayload): Promise<void> {
  const supabase = await createClient()
  const { error } = await supabase
    .from('users')
    .update({ membership_status: 'expired' })
    .eq('email', body.data.customerEmail)
  if (error) throw new Error(`memberExpired update failed: ${error.message}`)
}

async function handleTierChange(body: MayarWebhookPayload): Promise<void> {
  const supabase = await createClient()
  const newTier = detectTier(body.data.productName)
  const { error } = await supabase
    .from('users')
    .update({
      membership_tier: newTier,
      lifetime_purchased: newTier === 'lifetime',
      membership_expires_at: newTier === 'lifetime' ? null : undefined,
    })
    .eq('email', body.data.customerEmail)
  if (error) throw new Error(`changeTierMemberRegistered update failed: ${error.message}`)
}

// ─── Main handler ─────────────────────────────────────────────────────────────

export async function POST(request: NextRequest) {
  // Layer 1: Token check — fail fast before touching anything (D-04)
  const token = request.nextUrl.searchParams.get('token')
  if (!token || token !== process.env.MAYAR_WEBHOOK_TOKEN) {
    return new Response('Unauthorized', { status: 401 })
  }

  // Parse body
  let body: MayarWebhookPayload
  try {
    body = await request.json()
  } catch {
    return new Response('Invalid JSON', { status: 400 })
  }

  const eventType = body['event.received']
  const transactionId = body.data?.id

  if (!transactionId) {
    return new Response('Missing data.id', { status: 400 })
  }

  // Layer 2: Cross-verify with Mayar "Get Transaction" API (D-05)
  const { valid, httpStatus } = await crossVerifyWithMayar(transactionId)
  if (!valid) {
    return new Response('Cross-verify failed', { status: httpStatus })
  }

  // Layer 3: Idempotency INSERT — claim the event before processing (D-06)
  // Using createClient() — note: webhook_events has no RLS so service role write policy
  // must allow this. If createClient() uses anon key for webhooks route, ensure RLS allows
  // service role writes on webhook_events (migration sets no RLS on webhook_events intentionally).
  const supabase = await createClient()
  const { error: insertError } = await supabase
    .from('webhook_events')
    .insert({
      mayar_event_id: transactionId,
      event_type: eventType,
      payload: body,
    })

  if (insertError) {
    if (insertError.code === '23505') {
      // Already processed — idempotent success (D-06)
      return new Response('Already processed', { status: 200 })
    }
    console.error('webhook_events insert failed:', insertError.code, insertError.message)
    return new Response('DB error', { status: 500 })
  }

  // Dispatch by event type (D-07)
  try {
    switch (eventType) {
      case 'membership.newMemberRegistered':
        await handleNewMember(body)
        break
      case 'payment.received':
        await handlePaymentReceived(body)
        break
      case 'membership.memberUnsubscribed':
        await handleUnsubscribed(body)
        break
      case 'membership.memberExpired':
        await handleExpired(body)
        break
      case 'membership.changeTierMemberRegistered':
        await handleTierChange(body)
        break
      default:
        // D-08: unknown event types — log and return 200 (do NOT error; Mayar would retry forever)
        console.log(`[mayar-webhook] unknown event type: ${eventType}`)
    }
  } catch (err) {
    console.error('[mayar-webhook] dispatch error:', err)
    // Return 500 so Mayar retries — the idempotency ledger INSERT already succeeded,
    // so retry will hit the 23505 UNIQUE violation and return 200 immediately.
    // This means a retry after partial failure is safe.
    return new Response('Handler error', { status: 500 })
  }

  // Mark event as fully processed
  await supabase
    .from('webhook_events')
    .update({ processed_at: new Date().toISOString() })
    .eq('mayar_event_id', transactionId)

  return new Response('OK', { status: 200 })
}
```
